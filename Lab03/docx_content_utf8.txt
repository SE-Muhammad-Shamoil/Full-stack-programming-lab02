Manipulate HTML elements using DOM methods. JavaScript Implementation with ES6 features: Inline JS: Directly inside HTML tag. Example: &lt;button onclick=&quot;alert('Hi')&quot;&gt;Click&lt;/button&gt; Internal JS: Inside &lt;script&gt; tag. Example: &lt;script&gt;alert('Hello')&lt;/script&gt; External JS: Separate .js file linked to HTML. Example: &lt;script src=&quot;script.js&quot;&gt;&lt;/script&gt; let Block-scoped variable (works inside { } only). Example: &lt;body&gt; &lt;h3&gt; Full JavaScript ES6 Features Lab - Practical Implementation &lt;/h3&gt; &lt;h4&gt; let &lt;/h4&gt; &lt;p id = &quot;demo1&quot; &gt;&lt;/p&gt; &lt;script&gt; { let x = 10 ; document.getElementById( &quot;demo1&quot; ).innerHTML = x; } &lt;/script&gt; &lt;/body&gt; const Block-scoped variable that cannot be reassigned. Example: &lt;h4&gt; const &lt;/h4&gt; &lt;p id = &quot;demo2&quot; &gt;&lt;/p&gt; &lt;script&gt; const pi = 3.14 ; document.getElementById( &quot;demo2&quot; ).innerHTML = pi; &lt;/script&gt; Arrow Functions Short way to write functions using =&gt;. Example: &lt;h4&gt; Arrow Functions &lt;/h4&gt; &lt;p id = &quot;demo3&quot; &gt;&lt;/p&gt; &lt;script&gt; const greet = name =&gt; &quot;Hello &quot; + name; document.getElementById( &quot;demo3&quot; ).innerHTML = greet( &quot;Ali&quot; ); &lt;/script&gt; Template Literals Use backticks ` ` to insert variables inside strings. Example: &lt;h4&gt; Template Literals &lt;/h4&gt; &lt;p id = &quot;demo4&quot; &gt;&lt;/p&gt; &lt;script&gt; let name = &quot;Ahmed&quot; ; document.getElementById( &quot;demo4&quot; ).innerHTML = `Welcome ${ name } ` ; &lt;/script&gt; Rest Operator (...) Collects multiple values into an array. Example: &lt;h4&gt; Rest Operator (...) &lt;/h4&gt; &lt;p id = &quot;demo6&quot; &gt;&lt;/p&gt; &lt;script&gt; function add(...numbers) { return numbers[ 0 ] + numbers[ 1 ]; } document.getElementById( &quot;demo6&quot; ).innerHTML = add( 10 , 20 ); &lt;/script&gt; Spread Operator (...) Expands array values. Example : &lt;h4&gt; Spread Operator (...) &lt;/h4&gt; &lt;p id = &quot;demo7&quot; &gt;&lt;/p&gt; &lt;script&gt; let arr1 = [ 1 , 2 ]; let arr2 = [...arr1, 3 , 4 ]; document.getElementById( &quot;demo7&quot; ).innerHTML = arr2; &lt;/script&gt; Destructuring Extract values from arrays or objects easily. Example : &lt;h4&gt; Destructuring &lt;/h4&gt; &lt;p id = &quot;demo8&quot; &gt;&lt;/p&gt; &lt;script&gt; let person = { name: &quot;Ali&quot; , age: 20 }; let { name, age } = person; document.getElementById( &quot;demo8&quot; ).innerHTML = name + &quot; is &quot; + age; &lt;/script&gt; Classes Blueprint to create objects. Example : &lt;h4&gt; class &lt;/h4&gt; &lt;p id = &quot;demo9&quot; &gt;&lt;/p&gt; &lt;script&gt; class Student { constructor (name) { this .name = name; } } let s1 = new Student( &quot;Hassan&quot; ); document.getElementById( &quot;demo9&quot; ).innerHTML = s1.name; &lt;/script&gt; Promises Handle asynchronous operations and it is an object that handles tasks that take time (like loading data, API calls, timers). It represents a value that will be available now, later, or maybe never . Example : &lt;h4&gt; promise &lt;/h4&gt; &lt;h5 id = &quot;demo10&quot; &gt;&lt;/h5&gt; &lt;script&gt; let myPromise = new Promise( function (resolve, reject) { let success = true ; // change to false to see reject if (success) { resolve( &quot;Task Completed Successfully!&quot; ); } else { reject( &quot;Task Failed!&quot; ); } }); myPromise.then( function (result) { document.getElementById( &quot;demo10&quot; ).innerHTML = result; }).catch( function (error) { document.getElementById( &quot;demo10&quot; ).innerHTML = error; }); &lt;/script&gt; Map Stores key-value pairs. Example : &lt;h4&gt; map &lt;/h4&gt; &lt;p id = &quot;demo13&quot; &gt;&lt;/p&gt; &lt;script&gt; let map = new Map(); map.set( &quot;name&quot; , &quot;Ali&quot; ); document.getElementById( &quot;demo13&quot; ).innerHTML = map.get( &quot;name&quot; ); &lt;/script&gt; Set Stores unique values only. Example : &lt;h4&gt; set &lt;/h4&gt; &lt;p id = &quot;demo14&quot; &gt;&lt;/p&gt; &lt;script&gt; let set = new Set([ 1 , 2 , 2 , 3 ]); document.getElementById( &quot;demo14&quot; ).innerHTML = set.size; &lt;/script&gt; Generators Functions that pause using yield. Example : &lt;h4&gt; Generators &lt;/h4&gt; &lt;p id = &quot;demo16&quot; &gt;&lt;/p&gt; &lt;script&gt; function* gen() { yield 1 ; yield 2 ; } let g = gen(); document.getElementById( &quot;demo16&quot; ).innerHTML = g.next().value; &lt;/script&gt; JSON JSON (JavaScript Object Notation) is a way to store and send data. Think of it like a digital form . It’s just text , but it looks like a JavaScript object. Used everywhere in web apps to exchange data . Features of JSON Data is stored in key-value pairs Keys must be in double quotes &quot; &quot; Example: &lt;h4&gt; Demo json &lt;/h4&gt; &lt;h2 id = &quot;demojson&quot; &gt;&lt;/h2&gt; &lt;script&gt; // Step 1: Create a JavaScript object const student = { name: &quot;Ali&quot; , age: 22 , courses: [ &quot;JS&quot; , &quot;HTML&quot; , &quot;CSS&quot; ] }; // Step 2: Convert object to JSON string (to send to server) const jsonString = JSON.stringify(student); // Step 3: Convert JSON string back to object (received from server) const obj = JSON.parse(jsonString); // Step 4: Show in HTML document.getElementById( &quot;demojson&quot; ).innerHTML = `Name: ${ obj.name } &lt;br&gt; Age: ${ obj.age } &lt;br&gt; Courses: ${ obj.courses.join( &quot;, &quot; ) } ` ; &lt;/script&gt; Lab Task Instructions: Create a GitHub account if you don’t have one.
